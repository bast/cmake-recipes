# set minimum cmake version
cmake_minimum_required(VERSION 3.5 FATAL_ERROR)

# project name and language
project(recipe-04 CXX)

# list sources
set(_sources Message.hpp Message.cpp)

# introduce a toggle for using a library
set(USE_LIBRARY OFF)
# BUILD_SHARED_LIBS is a global flag offered by CMake to toggle the behavior of
# add_library
set(BUILD_SHARED_LIBS OFF)

if(USE_LIBRARY) # add_library will create a static library  since
                # BUILD_SHARED_LIBS is OFF
  add_library(message ${_sources})

  add_executable(hello-world hello-world.cpp)

  target_link_libraries(hello-world message)
else()
  add_executable(hello-world hello-world.cpp ${_sources})
endif()
